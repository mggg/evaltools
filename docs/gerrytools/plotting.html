13    """
14    Draws a gerrychain Graph object. Returns a single Axes object (for dual
15    graphs drawn whole) and lists of `(Figure, Axes)` pairs for graphs drawn
16    component-wise.
17
18    Args:
19        G (Graph): The dual graph to draw.
20        ax (Axes, optional): `matplotlib.axes.Axes` object. If not passed, one
21            is created.
22        x (str, optional): Vertex property used as the horizontal (E-W) coordinate.
23        y (str, optional): Vertex property used as the vertical (N-S) coordinate.
24        components (bool, optional): If `True`, the graph is assumed to have
25            more than one connected component (e.g. Michigan) and is drawn
26            component-wise and rather than return a single `Axes` object, return
27            a list of `(Figure, Axes)` pairs. If something is passed to `ax`, the
28            same Axes instance is used for each new Figure.
29        node_size (float, optional): Specifies the default size of a vertex.
30        kwargs (dict, optional): Arguments to be passed to `nx.draw()`.
31
32    Returns:
33        A tuple of `matplotlib` `(Figure, Axes)` objects, or if `components` is
34        `True`, returns a list of `(Figure, Axes)` objects corresponding to each
35        component.
36    """
37    # Create a mapping from identifiers to positions.
38    positions = {
39        v: (properties[x], properties[y]) for v, properties in G.nodes(data=True)
40    }
41
42    # If `components` is true, plot the graph component-wise. Otherwise plot
43    # normally. First, set some properties common to both graphs.
44    properties = {"pos": positions, "node_size": node_size}
45
46    # Initialize `pairs` to None.
47    pairs = None
48
49    if not components:
50        if not ax:
51            axes = plt.axes()
52        else:
53            axes = ax
54        nx.draw(G, ax=axes, **properties, **kwargs)
55    else:
56        # Create lists for figures and axes.
57        pairs = []
58
59        connected_components = [c for c in nx.connected_components(G)]
60        for component in connected_components:
61            # Create a new Figure object for each component.
62            fig = plt.figure()
63            if not ax:
64                ax = plt.axes()
65
66            # Plot the graph.
67            subgraph = G.subgraph(component)
68            nx.draw(subgraph, ax=ax, **properties, **kwargs)
69
70            # Add them to their respective lists.
71            pairs.append((fig, ax))
72
73    return pairs if pairs else axes

 10def violin(
 11    ax,
 12    scores,
 13    xticklabels=None,
 14    labels=None,
 15    proposed_info={},
 16    percentiles=(1, 99),
 17    rotation=0,
 18    ticksize=12,
 19    jitter=1 / 3,
 20) -> Axes:
 21    r"""
 22    Plot a violin plot, which takes `scores` — a dictionary where each value (corresponding to
 23    an ensemble, citizens' ensemble, or proposed plans), will be a list of lists, where each
 24    sublist will be its own violin. Proposed scores will be plotted as colored circles on
 25    their respective violin. Color the violins conditioned on the kind of the scores
 26    (ensemble or citizen), and trim each sublist to only the values between the percentiles.
 27
 28    Args:
 29        ax (Axes): `Axes` object on which the violins are plotted.
 30        scores (dict): Dictionary with keys of `ensemble`, `citizen`, `proposed`
 31            which map to lists of numerical scores.
 32        proposed_info (dict, optional): Dictionary with keys of `colors`, `names`;
 33            the \(i\)th color in `color` corresponds to the \(i\)th name in `names.
 34        percentiles (tuple, optional): Observations outside this range of
 35            percentiles are ignored. Defaults to `(1, 99)`, such that observations
 36            between the 1st and 99th percentiles (inclusive) are included, and
 37            all others are ignored.
 38        rotation (float, optional): Tick labels are rotated `rotation` degrees
 39            _counterclockwise_.
 40        ticksize (float, optional): Font size for tick labels.
 41        jitter (float, optional): When there is more than one proposed plan,
 42            adjust its detail points by a value drawn from \(\mathcal U (-\epsilon,
 43            \epsilon)\) where \(\epsilon = \) `jitter`.
 44        labels (list, optional): x- and y-axis labels, if desired.
 45        xticklabels (list, optional): Labels for the violins, default to integers.
 46
 47    Returns:
 48        `Axes` object on which the violins are plotted.
 49    """
 50    # Get all the scores into one list; pick a face color.
 51    ensemble = scores["ensemble"] if scores["ensemble"] else scores["citizen"]
 52    facecolor = defaultGray if scores["ensemble"] else citizenBlue
 53
 54    # Initialize a list for winnowing scores.
 55    trimmed_scores = []
 56
 57    # Pare each ensemble down to only the observations between the 1st and 99th
 58    # percentiles.
 59    for score_list in ensemble:
 60        low = np.percentile(ensemble, percentiles[0])
 61        high = np.percentile(ensemble, percentiles[1])
 62        # print(f"Only including scores between [{low}, {high}]")
 63        trimmed_scores.append([s for s in score_list if s >= low and s <= high])
 64
 65    # Plot violins.
 66    parts = ax.violinplot(trimmed_scores, showextrema=False)
 67
 68    # For each of the violins, modify its visual properties; change the face color
 69    # to the specified face color, change its edge color to black, and set its
 70    # opacity to 1.
 71    for pc in parts["bodies"]:
 72        pc.set_facecolor(facecolor)
 73        pc.set_edgecolor("black")
 74        pc.set_alpha(1)
 75
 76    # Set xticks, xlabels, and x-axis limits.
 77    if not xticklabels:
 78        xticklabels = range(1, len(scores["ensemble"]) + 1)
 79    ax.set_xticks(range(1, len(ensemble) + 1))
 80    ax.set_xticklabels(xticklabels, fontsize=ticksize, rotation=rotation)
 81    ax.set_xlim(0.5, len(ensemble) + 0.5)
 82
 83    # Plot each proposed plan individually, adjusting its detail points by
 84    # a value drawn from the uniform distribution of specified width centered on
 85    # the index of the violin.
 86    if scores["proposed"]:
 87        for violin in range(len(scores["proposed"])):
 88            for plan, score in enumerate(scores["proposed"][violin]):
 89                # Horizontally jitter proposed scores if there are multiple scores
 90                # at the same height.
 91                jitter_val = (
 92                    random.uniform(-jitter, jitter)
 93                    if scores["proposed"][violin].count(score) > 1
 94                    else 0
 95                )
 96                ax.scatter(
 97                    violin + 1 + jitter_val,
 98                    score,
 99                    color=scores["proposed"]["colors"][violin]
100                    if scores["proposed"]["colors"]
101                    else districtr(plan + 1).pop(),
102                    edgecolor="black",
103                    s=100,
104                    alpha=0.9,
105                    label=proposed_info["names"][plan] if violin == 0 else None,
106                )
107        ax.legend()
108        ax.grid(axis="x")
109
110    if labels:
111        ax.set_xlabel(labels[0], fontsize=24)
112        ax.set_ylabel(labels[1], fontsize=24)
113
114    return ax
