  7class AssignmentCompressor:
  8    """
  9    A class for compressing and decompressing lots of assignments very, very
 10    quickly. Intended for use with ``jsonlines``-like libraries (where
 11    assignments are read in line-by-line) or for network requests (where
 12    assignments are retrieved one-by-one). When decompressing, yields ``dict``s
 13    where keys are in sorted order.
 14
 15    The compression schema considers the set of unique identifiers, imposes an
 16    ordering (lexicographic order) on the identifiers, and matches the
 17    assignment to that ordering. We assign all unassigned units to ``"-1"``
 18    and, once the default cache size is hit (or assignments are no longer being
 19    read in), compress all assignments in the cache. Assignments are read in
 20    and out in the same order, and the keys for each assignment are in the
 21    same order.
 22
 23    Example:
 24        To compress assignments, we need a set of unique identifiers such that
 25        each identifier maps one geometric unit to one district.
 26
 27            ...
 28
 29            geoids = blocks["GEOID20"].astype(str)
 30            ac = AssignmentCompressor(
 31                 geoids, location="compressed-assignments.ac")
 32
 33            with ac as compressor:
 34                for assignment in assignments:
 35                    # Here, ensure that all assignments have string keys and
 36                    # string values; also ensure that an assignment's keys are
 37                    # a subset of geoids (or whatever IDs you're passing).
 38                    compressor.compress(assignment)
 39
 40            ...
 41
 42        To decompress assignments, we again must have a set of unique geometric
 43        identifiers which match the assignments. We can then iterate over the
 44        decompressed assignments as they're read out of the file.
 45
 46            ...
 47
 48            geoids = blocks["GEOID20"].astype(str)
 49            ac = AssignmentCompressor(geoids, location="compressed-assignments.ac")
 50
 51            for assignment in ac.decompress():
 52                <do whatever!>
 53
 54            ...
 55
 56    Attributes:
 57        DISTRICT_DELIMITER (bytes): A bytestring which separates district
 58            identifiers in an assignment.
 59        ASSIGNMENT_DELIMITER (bytes): A bytestring which separates assignments from
 60            each other.
 61        CHUNK_DELIMITER (bytes): A bytestring which separates assignment chunks
 62            from each other.
 63        CHUNK_SIZE (int): Default number of bytes read in from the IO stream at each
 64            step.
 65        ENCODING (str): Default string encoding style.
 66        identifiers: A sortable, iterable collection of unique items corresponding
 67            to geographic identifiers.
 68        compressed: A pandas `Index` containing the identifiers; this is used
 69            to quickly perform vectorized identifier matchings, rather than using
 70            traditional iterative methods.
 71        cache: Collection of assignments to be compressed. Assignments are loaded
 72            into the cache every time the ``.compress()`` method is called, and
 73            is cleared whenever the length of the cache exceeds the window width.
 74        window: Maximum cache length before the cache is compressed, written to
 75            file, and emptied.
 76        default: The default assignment which is updated each time an assignment
 77            is passed to the compressor.
 78        location: The place to which compressed data is written or read.
 79    """

 81    def __init__(self, identifiers, window=10, location="compressed.ac"):
 82        """
 83        Creates `AssignmentCompressor` instance.
 84
 85        Args:
 86            identifiers (list): An iterable collection of string identifiers; any
 87                assignment's keys must be a subset of `identifiers`.
 88            window (int, optional): A positive integer representing the cache
 89                window size. Defaults to cache.
 90            location (str, optional): The path to the compressed resource (read
 91                or write). Defaults to `compressed.ac`.
 92        """
 93        self.identifiers = SortedList(identifiers)
 94        self.default = frozenset(zip(self.identifiers, ["-1"] * len(self.identifiers)))
 95        self.cache = []
 96        self.location = location
 97
 98        # Error to users if the window is nonexistent.
 99        if not isinstance(window, int) or window <= 0:
100            raise ValueError("Cache window width must be a positive integer.")
101
102        self.window = window
103
104        self.DISTRICT_DELIMITER = b","
105        self.ASSIGNMENT_DELIMITER = b"<<<*>>>"
106        self.CHUNK_DELIMITER = b"(((*)))"
107        self.CHUNK_SIZE = 16384
108        self.ENCODING = "raw_unicode_escape"

110    def match(self, assignment) -> SortedDict:
111        """
112        Matches an assignment to an index (the set of geometric identifiers)
113        and returns a `SortedDict`.
114
115        Args:
116            assignment (dict): Dictionary which matches geometric identifiers to
117                districts. All keys and values in this dictionary must be strings.
118
119        Returns:
120            A `SortedDict` with identifiers matched ti district assignments.s
121
122        """
123        # Create a dictionary which maps identifiers to `-1`, and update our
124        # dictionary with assignment values.
125        indexer = SortedDict(self.default)
126        indexer.update(assignment)
127
128        return indexer

153                to districts. All keys and values in these dictionaries must be

162    def compress(self, assignment):
163        """
164        Compresses the assignment `assignment` using `zlib`.
165
166        Args:
167            assignment (dict): Dictionary which matches geometric identifiers to districts.
168                All keys and values in this dictionary must be strings.
169        """
170        # If the user provides an empty assignment or the assignment's keys aren't
171        # a subset of `identifiers`, warn the user and skip the assignment.
172        skip = False
173
174        if not assignment:
175            skip = True
176            print("`assignment` is empty; skipping.")
177
178        if not set(assignment.keys()).issubset(self.identifiers):
179            skip = True
180            print(
181                "`assignment`'s keys are not a subset of `identifiers`; skipping. "
182                + "Please ensure that all keys and values in `assignment` are strings.",
183            )
184
185        # Join the things on the district separator, encode the whole thing, and
186        # encode according to the default encoding.
187        if not skip:
188            indexed = self.match(assignment)
189            sep = self.DISTRICT_DELIMITER.decode()
190            encoded = bytes(sep.join(indexed.values()).encode(self.ENCODING))
191
192            # Compress.
193            self.cache += [encoded]
194            self._compress()
195
196    def _compress(self, force=False):
197        """
198        Private method which actually does the compression.
199
200        Args:
201            force: If truthy, then the length of the cache is ignored, the data
202                in the cache are compressed, and the compressed data is written
203                to file. ``force`` is truthy when teardown logic is entered
204                (i.e. after ``__exit()__`` is called).
205        """
206        # Check to see if the cache is full. If so, compress the data, write to
207        # file, and reset the cache.
208        if len(self.cache) == self.window or force:
209            with open(self.location, "ab") as writer:
210                # Write compressed data to file.
211                compressed = zlib.compress(
212                    bytes(self.ASSIGNMENT_DELIMITER.join(self.cache))
213                )
214                writer.write(compressed)
215
216                # We only forcibly write the chunk separator to file if we've
217                # entered teardown logic and the cache *is not empty*. If the
218                # cache is empty when we're entering teardown logic, that means
219                # (number of assignments compressed) == (window width), in which
220                # case we've reached the end of compression and should not write
221                # a separator; doing so will produce an empty bytestring (which,
222                # in turn, produces a dictionary with one key, corresponding to
223                # a null assignment).
224                if not force:
225                    writer.write(self.CHUNK_DELIMITER)
226
227            # Reset the cache.
228            self.cache = []
229
230    def decompress(self):
231        """
232        Decompresses the data at ``location``. A generator which ``yield``s
233        assignments.
234
235        Yields:
236            Decompressed assignment dictionaries.
237        """
238        # Open the compressed file. Then we read it in chunks, loading until
239        # we hit our separator or until the end of the file.
240        with open(self.location, "rb") as _compressed_fin:
241            for chunk in self._chunk(_compressed_fin):
242                if not chunk:
243                    break
244                for assignment in self._decompress(chunk):
245                    yield assignment
246
247    def _chunk(self, stream):
248        """
249        Private method for reading chunks from file without holding the entire
250        file in memory. A generator for decompressed assignments.
251
252        Args:
253            stream: A ``BytesIO`` instance from which data is read.
254
255        Yields:
256            Buffered, compressed bytes to be fed into the decompressor.
257        """
258        # Create a buffer.
259        _buffer = []
260
261        # Read until we hit the end of the file `yield`ing each chunk as we go.
262        while True:
263            # Read in a chunk of data.
264            chunk = stream.read(self.CHUNK_SIZE)
265            _buffer.append(chunk)
266
267            # Check if the chunk has our delimiter in it. If it contains our
268            # delimiter, then the buffer *up to the delimiter* contains compressed
269            # assignments; this should be `yield`ed and decompressed. We only
270            # want to get the part before the delimiter for decompression, but
271            # retain the rest.
272            if self.CHUNK_DELIMITER in chunk:
273                _buffered_bytes = b"".join(_buffer)
274                part, _buffer_first = _buffered_bytes.split(self.CHUNK_DELIMITER, 1)
275                _buffer = [_buffer_first]
276                yield part
277
278            # If the chunk's empty, `yield` the remaining buffer and return.
279            if not chunk:
280                yield b"".join(_buffer)
281                break
282
283    def _decompress(self, chunk) -> List[dict]:
284        """
285        Private method which decompresses assignments.
286
287        Args:
288            chunk: Compressed, byte-encoded data representing ``window``
289                assignments.
290
291        Returns:
292            List of decompressed assignment objects.
293        """
294        # Decompress the chunk and split it on our delimiter.
295        decompressed = zlib.decompress(chunk)
296        decompressed_parts = decompressed.split(self.ASSIGNMENT_DELIMITER)
297
298        # For each of the parts, decode the bytes, make them into lists, and
299        # match them to GEOIDs.
300        decoded_parts = [part.decode() for part in decompressed_parts]
301        split_parts = [
302            part.split(self.DISTRICT_DELIMITER.decode()) for part in decoded_parts
303        ]
304        indexed_parts = [dict(zip(self.identifiers, part)) for part in split_parts]
305
306        return indexed_parts

A class for compressing and decompressing lots of assignments very, very quickly. Intended for use with jsonlines-like libraries (where assignments are read in line-by-line) or for network requests (where assignments are retrieved one-by-one). When decompressing, yields dicts where keys are in sorted order.

The compression schema considers the set of unique identifiers, imposes an ordering (lexicographic order) on the identifiers, and matches the assignment to that ordering. We assign all unassigned units to "-1" and, once the default cache size is hit (or assignments are no longer being read in), compress all assignments in the cache. Assignments are read in and out in the same order, and the keys for each assignment are in the same order.
