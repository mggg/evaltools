def splits( unit: str, names: bool = False, popcol: str = None, how: str = 'pandas', alias: str = None) -> gerrytools.scoring.types.Score: 
162def splits(
163    unit: str,
164    names: bool = False,
165    popcol: str = None,
166    how: str = "pandas",
167    alias: str = None,
168) -> Score:
169    """
170    Score representing the number of units split by the districting plan.
171
172    Bear in mind that this calculates the number of *unit splits*, not the number
173    of *units split*: for example, if a district divides a county into three
174    pieces, the former reports two splits (as a unit divided into three pieces is
175    cut twice), while the latter would report one split (as there is one county
176    being split).
177
178    Args:
179        unit (str): Data column; each assigns a vertex to a unit.
180            Generally, these units are counties, VTDs, precincts, etc.
181        popcol (str, optional): The population column on the `Partition`'s dual
182            graph. If this is passed, then a unit is only considered "split" if
183            the _populated_ base units end up in different districts.
184        how (str, optional): How do we perform these calculations on the back
185            end? Acceptable values are `"pandas"` and `"gerrychain"`; defaults to
186            `"pandas"`.
187        names (bool, optional): Whether we return the identifiers of the things
188            being split.
189
190    Returns:
191        A score object with the name `"{alias}_splits"` and associated function that takes a
192        partition and returns a PlanWideScoreValue for the number of splits.
193    """
194    if alias is None:
195        alias = unit
196
197    return Score(
198        f"{alias}_splits",
199        partial(_splits, unit=unit, how=how, popcol=popcol, names=names),
200    )

42def summarize(
43    part: Partition,
44    scores: Iterable[Score],
45    gdf: Optional[GeoDataFrame] = None,
46    join_on: Optional[str] = None,
47) -> Dict[str, ScoreValue]:
48    """
49    Summarize the given partition by the passed scores.
50
51    Args:
52        part (Partition): The plan to summarize.
53        scores (Iterable[Score]): Which scores to include in the summary.
54        gdf (GeoDataFrame): Geometries of nodes in the dual graph used by `part`.
55            Only necessary when using scoring functions that rely on dissolved
56            district geometries (most geometric scoring functions).
57        join_on (str): Field used to join `part.graph` to `gdf`.
58            If not specified, geometries are joined by matching the index of `gdf`
59            to the node keys of `part.graph`.
60
61    Raises:
62        ValueError: If `gdf` is not specified and at least one score in `scores`
63            is dissolved.
64
65    Returns:
66        A dictionary that maps score names to the corresponding ScoreValues of the score functions
67        applied to the plan.
68
69        ie.
70        `{"cut_edges": 4050, "num_party_seats": 3, ... }`
71    """
72    if any(score.dissolved for score in scores):
73        if gdf is None:
74            raise ValueError("Geometries must be provided for dissolved scores.")
75
76        if join_on is None:
77            assignment = dict(part.assignment)
78            gdf = gdf.copy(deep=False)
79        else:
80            assignment = {
81                part.graph.nodes[node][join_on]: label
82                for node, label in part.assignment.items()
83            }
84            gdf = gdf.set_index(join_on)
85
86        gdf["assignment"] = assignment
87        dissolved_gdf = gdf.dissolve(by="assignment")
88    else:
89        dissolved_gdf = None
90
91    summary = {}
92    for score in scores:
93        if score.dissolved:
94            summary[score.name] = score.apply(dissolved_gdf)
95        else:
96            summary[score.name] = score.apply(part)
97    return summary
